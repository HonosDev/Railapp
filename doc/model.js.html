 function isApiDate(dateToCheck){
 	if(dateToCheck.match("([0-9]{4})([-])([0-9]{2})([-])([0-9]{2})")){
 		return true;
 	}else{
 		return false;
 	}
 }

 function toApiDate(dateToConvert){
 	var split = dateToConvert.split(/([0-9]{2}).([0-9]{2}).([0-9]{4})/);
 	return split[3]+"-"+split[1]+"-"+split[2];
 }

/**
 * Station to Code allow you to get the code of a station 
 */
function stationToCode(station){
	if(stations[station] == undefined){
		return "Unk station";
	}else{
		return stations[station];
	}
}

/**
 * Code to station allow you to get the station using the station code
 */
function codeToStation(code){
	for(var row in stations){
		if(stations[row] == code){
			return row;
		}
	}
}
